def []=(key_p, val_p)
  key = find_param(key_p)
  if (key.nil?)
    raise ArgumentError.new('unknown query or path parameter ' +
                            key_p.inspect +
                            " for class #{self.class.name}")
  end
  return self.__send__("#{key}=".to_sym, val_p.to_s)
end
