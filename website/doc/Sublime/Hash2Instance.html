def initialize(*args_p)
 args_p.select { |o| o.kind_of?(Hash) }.each do |hsh_p|
    hsh_p.each do |k,v|
      ivwriter	= (k.to_s + '=').to_sym
      ivname	= ('@' + k.to_s).to_sym
      if (self.class.method_defined?(ivwriter))
        self.send(ivwriter, v)
      else
        self.instance_variable_set(ivname, v)
      end
    end
  end
end
